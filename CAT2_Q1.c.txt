#include<stdio.h>

int main(){
    int i; 
    int j;
    int k;
    int l;
    int score [2][2]= {
        {65,92},
        {84,72}
    };
    int score2[2][2]={
        {35,70},
        {59,67}
    };
    printf("SCORES\n");
    for (i=0;i<2;i++){
        for(j=0;j<2;j++){
            printf("%d\t", score [i][j]);
        }
        printf("\n");
    }
    printf("\nSCORES\n");
    for(k=0;k<2;k++){
        for(l=0;l<2;l++){
            printf("%d\t",score2[k][l]);
        }
        printf("\n");
    }
    return 0;
}
